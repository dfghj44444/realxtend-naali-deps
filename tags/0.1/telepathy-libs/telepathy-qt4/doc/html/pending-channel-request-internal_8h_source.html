00032 class PendingChannelRequestCancelOperation : public PendingOperation
00033 {
00034     Q_OBJECT
00035     Q_DISABLE_COPY(PendingChannelRequestCancelOperation)
00036 
00037 public:
00038     PendingChannelRequestCancelOperation(QObject *parent)
00039         : PendingOperation(parent)
00040     {
00041     }
00042 
00043     ~PendingChannelRequestCancelOperation()
00044     {
00045     }
00046 
00047     void go(const ChannelRequestPtr &channelRequest)
00048     {
00049         Q_ASSERT(mChannelRequest.isNull());
00050         mChannelRequest = channelRequest;
00051         connect(mChannelRequest->cancel(),
00052                 SIGNAL(finished(Tp::PendingOperation*)),
00053                 SLOT(onCancelOperationFinished(Tp::PendingOperation*)));
00054     }
00055 
00056 private Q_SLOTS:
00057     void onCancelOperationFinished(Tp::PendingOperation *op)
00058     {
00059         if (op->isError()) {
00060             setFinishedWithError(op->errorName(), op->errorMessage());
00061             return;
00062         }
00063         setFinished();
00064     }
00065 
00066 private:
00067     ChannelRequestPtr mChannelRequest;
00068 };
